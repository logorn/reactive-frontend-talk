Callback hell

We're not handling errors

and if we would

We'd need to handle them separately

We can't compose this callbacks easily

There's no separation of concerns.

Let's see the Reactive way of doing this

$('.counter-btn).clicks()
.flatMap(function() {
  return Observable.fromPromise(apiServer.getCounterAmount(new Date()) 
}
.doOnEach(apiServer.logCounter)
.sum()
.subscribe(function(sum) {
  console.log("Count now is" , sum)
}, function error(e) {
  console.error("Error running this", e);
});

angular.module('counter', ['rx']);

angular.module('counter').controller('MainCtrl', 
  function(ApiServer, rx, $scope) {
      $scope.$createObservableFunction('increaseCounter')
      .flatMap(function() {
          return rx.Observable.fromPromise(ApiServer.getCounterAmount(new Date()));
      })
      .do(ApiServer.logCounter)
      .scan(function(acc, val) {return acc + val;})
      .subscribe(function(counter) {
          $scope.counter = counter;
          console.log("Total counter is", counter);
      }, function(error) {
          console.error("There was an error");
      });
});

$scope.$createObservableFunction('search')
  .map(function() {
      return $scope.name;
  })
  .flatMap(function(name) {
      return rx.Observable.fromPromise(ApiServer.findByName(name));
  })
  .filter(function(contact) {
      return !$scope.gender || ct.gender === $scope.gender;
  })
  .subscribe(function(people) {
      $scope.people = people;
  }, function(error) {
      console.error("There was an error");
